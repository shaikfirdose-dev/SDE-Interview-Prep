Question:

Take as input a number n, the number of rows  
Print the following pattern  
*** ***  
**   **  
*     *  
**   **  
*** ***  
for n = 5.  


Solution:

import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        
        int nstr = (n/2)+1; // initial no. of stars in forst row
        int nsp = 1;        // initial no.of spaces in first row
        
        for(int i=1;i<=n;i++){         // this loop is used for iterating the rows
            for(int j=1;j<=nstr;j++){   // this loop is used for stars
                System.out.print("*");
            }
            for(int k=1;k<=nsp;k++){   // this loop is used for printing the spaces
                System.out.print(" ");
            }
            for(int j=1;j<=nstr;j++){       // this loop is used for stars
                System.out.print("*");
            }
            
            if(i<=n/2){     // this steps for preparation of next iteration for upper part
                nsp=nsp+2;  // no.of spaces incremented by 2
                nstr--;     // no.of stars decremented by 1
            }       
            else{               // lower part
                nsp=nsp-2;   // no.of spaces decremented by 2
                nstr++;      // no.of stars incremented by 1
            }
            System.out.println();
        }
    }
}
