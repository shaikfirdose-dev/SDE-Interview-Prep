Question:

Take as input a number n, the number of rows
Print the following pattern
  *
 ***
*****
 ***
  *
for n = 5.

Solution:

import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        
        int nstr = 1; // initial no. of stars in first row
        int nsp = n/2; // initial no. of spaces in first row

        for(int i=1;i<=n;i++){   // this loop is for iterating the rows
            for(int j=1;j<=nsp;j++){ // this loop is for printing the spaces
                System.out.print(" ");
            }
            for(int k=1;k<=nstr;k++){ // this loop is for printing the no.of stars in a row
                System.out.print("*");
            }
            for(int j=1;j<=nsp;j++){ // this loop is for printing the no.of spaces
                System.out.print(" ");
            }

            if(i<=n/2){   	// this condition is for upper part
                nstr=nstr+2;	// updating the stars by 2
                nsp--;		// decrementing the spaces by 1
            }

            else{		// this condition is for lower part
                nstr=nstr-2;	// decrementing the stars by 2
                nsp++;		// incrementing the spaces by 1
            }
            System.out.println();  // going for next line
        }
    }
}
